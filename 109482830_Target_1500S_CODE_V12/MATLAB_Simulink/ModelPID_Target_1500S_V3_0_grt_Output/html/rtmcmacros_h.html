15/* Real-time Model Common Data Structure */



16typedef struct {



17  const char            **errorStatus;



18  RTWLogInfo            **rtwLogInfo;



19  RTWExtModeInfo        **extModeInfo;



20  RTWRTModelMethodsInfo *modelMethodsInfo;



21  RTWSolverInfo         **solverInfo;

23  time_T                **t;



24  void                  **timingData;

26  int_T                 **sampleHits;



27  int_T                 **perTaskSampleHitsPtr;

31} rtModelCommon;

42    MC.simTimeStep          = &rtmGetSimTimeStep(M);           \



43    MC.sampleHits           = &rtmGetSampleHitPtr(M);          \
