66//



67// ReadUINT8 : Reads a generic 8-bit value from the data area (used by public read methods)



68//



69bool CODK_CpuReadData::ReadUINT8(const long  byteOffset, // Buffer index to the data



70                                 ODK_UINT8  &value)      // Variable to store the read value



71{



72    // Verify offset is within data area



73    if ((byteOffset < 0) || ((byteOffset + ODK_SIZEBIT8) > m_nBytes))



74    {



75        return false;



76    }



77



78    // Read, convert, and return the data



79    value = reinterpret_cast<ODK_UINT8&>(m_data[byteOffset]);



80



81    return true;



82}

84//



85// ReadUINT16 : Reads a generic 16-bit value from the data area. Used by public read methods



86//



87bool CODK_CpuReadData::ReadUINT16(const long  byteOffset, // Buffer index to the data



88                                  ODK_UINT16 &value)      // Variable to store the read value



89{



90    // Verify offset is within data area



91    if ((byteOffset < 0) || ((byteOffset + ODK_SIZEBIT16) > m_nBytes))



92    {



93        return false;



94    }



95



96    // Read, convert, and return the data



97    ODK_UINT16 word = reinterpret_cast<ODK_UINT16&>(m_data[byteOffset]);



98    value = (((word & 0x00FFU) << 8) |



99             ((word & 0xFF00U) >> 8));



100



101    return true;



102}

152//



153// ReadS7BOOL : Reads a boolean value from the data area



154//



155bool CODK_CpuReadData::ReadS7BOOL(const long  byteOffset, // Buffer index to the data



156                                  int         bitNo,      // Bit number within the byte to read



157                                  bool       &value)      // Variable to store the read value



158{



159    ODK_UINT8 temp;



160



161    // Verify valid bit index



162    if ((bitNo < 0) || (bitNo > 7))



163    {



164        return false;



165    }



166



167    // Verify offset is within data area



168    if ((byteOffset < 0) || ((byteOffset + ODK_SIZEBIT8) > m_nBytes))



169    {



170        return false;



171    }



172



173    // read the byte containing the bit



174    if (!ReadUINT8(byteOffset, temp))



175    {



176        return false;



177    }



178



179    // Shift the bit to the first location



180    temp = static_cast<ODK_UINT8>(temp >> bitNo);



181



182    // Isolate the value



183    if ((temp & 1) == 1)



184    {



185        value = true;



186    }



187    else



188    {



189        value = false;



190    }



191    return true;



192}

250//



251// ReadS7DTL : Reads a Date and Time as DTL from the data area



252//



253bool CODK_CpuReadData::ReadS7DTL(long     byteOffset, // Offset to begin reading



254                                 ODK_DTL &dtl)        // DTL structure read



255{



256    // Verify offset is within data area



257    if ((byteOffset < 0) || ((byteOffset + static_cast<long>(sizeof(dtl))) > m_nBytes))



258    {



259        return false;



260    }



261



262    // year



263    ReadUINT16(byteOffset, dtl.Year);



264    byteOffset += ODK_SIZEBIT16;



265    // month



266    ReadUINT8(byteOffset, dtl.Month);



267    byteOffset += ODK_SIZEBIT8;



268    // day



269    ReadUINT8(byteOffset, dtl.Day);



270    byteOffset += ODK_SIZEBIT8;



271    // weekday



272    ReadUINT8(byteOffset, dtl.Weekday);



273    byteOffset += ODK_SIZEBIT8;



274    // hour



275    ReadUINT8(byteOffset, dtl.Hour);



276    byteOffset += ODK_SIZEBIT8;



277    // minute



278    ReadUINT8(byteOffset, dtl.Minute);



279    byteOffset += ODK_SIZEBIT8;



280    // second



281    ReadUINT8(byteOffset, dtl.Second);



282    byteOffset += ODK_SIZEBIT8;



283    // nanosecond



284    ReadUINT32(byteOffset, dtl.Nanosecond);



285



286	return CheckS7DTL(dtl);



287}

304//



305// ReadS7STRING : Reads an S7 string from the data area, and returns it as a C++ character string



306//                Truncate destination string, when source string is too long.



307//



308bool CODK_CpuReadData::ReadS7STRING(const long  byteOffset, // Buffer index to the data



309                                    ODK_UINT8   readMax,    // Maximum characters to read including the string terminating zero



310                                    ODK_CHAR*   string)     // Buffer to receive the string characters



311{



312    ODK_UINT8  curLen  = 0; // Current length of the string



313    ODK_UINT8  maxLen  = 0; // Maximum length of the string



314    ODK_UINT8  stop    = 0; // Maximum bytes to read



315    ODK_UINT16 strData = static_cast<ODK_UINT16>(byteOffset + ODK_SIZEBIT16); // Offset to the string data (characters)



316



317    // must receive a valid buffer for the string  AND  must have space for 1 character and the string terminator



318    if (0 == string  ||  readMax < 2)



319    {



320        return false;



321    }



322



323    // Get the current string length



324    if (! ReadS7STRING_LEN(byteOffset, maxLen, curLen))



325    {



326        return false;



327    }



328



329    // Verify offset not out of bounds



330    if ((byteOffset < 0) || (byteOffset + static_cast<long>(maxLen) + ODK_SIZEBIT16 > m_nBytes))



331    {



332        return false;



333    }



334



335    // Get the value to read to



336    if (curLen < readMax)



337    {



338        stop = curLen;



339    }



340    else



341    {



342        stop = static_cast<ODK_UINT8>(readMax - 1);



343    }



344



345    // Read the value



346    memcpy(string, &(m_data[strData]), stop);



347    string[stop] = '\0';  // terminate the string



348



349    return true;



350}

352// return true, when DTL is in valid range



353bool CODK_CpuReadData::CheckS7DTL(const ODK_DTL& dtl) const



354{



355	// year



356    if (1970 > dtl.Year  ||  2262 < dtl.Year)



357    {



358        return false;



359    }



360    // month



361    if (1 > dtl.Month  ||  12 < dtl.Month)



362    {



363        return false;



364    }



365    // day



366    if (1 > dtl.Day  ||  31 < dtl.Day)



367    {



368        return false;



369    }



370    // weekday



371    if (1 > dtl.Weekday  ||  7 < dtl.Weekday)



372    {



373        return false;



374    }



375    // hour



376    if (23 < dtl.Hour)



377    {



378        return false;



379    }



380    // minute



381    if (59 < dtl.Minute)



382    {



383        return false;



384    }



385    // second



386    if (59 < dtl.Second)



387    {



388        return false;



389    }



390    // nanosecond



391    if (999999999 < dtl.Nanosecond)



392    {



393        return false;



394    }



395



396    return true;



397}

399// return true, when S5Time is in valid range



400bool CODK_CpuReadData::CheckS7S5Time(const ODK_UINT16& value) const



401{



402    ODK_INT8 bcd;



403



404    // check time value



405    bcd = GetBCD(value & 0xFF);



406    if (bcd < 0  ||  bcd > 99) return false;



407    bcd = GetBCD((value >> 8) & 0x0F);



408    if (bcd < 0  ||  bcd > 9) return false;



409    // check time base is always valid (two bits, four states)



410



411    return true;



412}

414// return true, when DT is in valid range



415bool CODK_CpuReadData::CheckS7DateAndTime(const ODK_UINT64& value) const



416{



417    ODK_INT8 bcd;



418



419    // check year



420    bcd = GetBCD((value >> 56) & 0xFF);



421    if (bcd < 0  ||  bcd > 99) return false;



422    // check month



423    bcd = GetBCD((value >> 48) & 0xFF);



424    if (bcd < 1 || bcd > 12) return false;



425    // check day



426    bcd = GetBCD((value >> 40) & 0xFF);



427    if (bcd < 1 || bcd > 31) return false;



428    // check hour



429    bcd = GetBCD((value >> 32) & 0xFF);



430    if (bcd < 0 || bcd > 23) return false;



431    // check minute



432    bcd = GetBCD((value >> 24) & 0xFF);



433    if (bcd < 0 || bcd > 59) return false;



434    // check second



435    bcd = GetBCD((value >> 16) & 0xFF);



436    if (bcd < 0 || bcd > 59) return false;



437    // check millisecond



438    bcd = GetBCD((value >> 8) & 0xFF);



439    if (bcd < 0 || bcd > 99) return false;



440    bcd = GetBCD((value >> 4) & 0x0F);



441    if (bcd < 0 || bcd > 9) return false;



442    // check weekday



443    bcd = GetBCD(value & 0x0F);



444    if (bcd < 1 || bcd > 7) return false;



445



446    return true;



447}

449// return bcd number of value, or -1 when value is no BCD number



450ODK_INT8 CODK_CpuReadData::GetBCD(const ODK_UINT8 value) const



451{



452    if ((value & 0x0F) > 9) return -1;



453    if (((value >> 4) & 0x0F) > 9) return -1;



454



455    return ((value >> 4) * 10) + (value & 0x0F);



456}
