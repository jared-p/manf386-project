7/*



8 * Maps out a section of a circular buffer.  When the buffer is not



9 * wrapped, only section 1 is required.  When the buffer is wrapped



10 * both sections are required.



11 */



12typedef struct BufMem_tag {



13    int_T   nBytes1;    /* nbytes in sections 1          */



14    char_T  *section1;  /* ptr to start of section 1     */



15



16    int_T   nBytes2;    /* nbytes in section 2 (0 if not wrapped)          */



17    char_T  *section2;  /* ptr to start of section 2 (NULL if not wrapped) */



18} BufMem;

22/*



23 * For each of nActiveBufs (buffers with data in them) we have a list of



24 * the buffer memory (bufs) and a list of the tid's with which this buffer



25 * is associated.



26 */



27typedef struct ExtBufMemList_tag {



28    int_T        nActiveBufs; /* num bufs with data to upload */



29



30    const BufMem *bufs; /* sections of each buffer for uploading */



31    const int_T  *tids; /* tid associated with each section      */



32} ExtBufMemList;
