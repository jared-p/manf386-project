1/* cg_sfun.h 



2 *  



3 * Entry point for user-written S-Functions when compiled for



4 * use with the Simulink C-Code Generator.  This file should



5 * be conditionally included at the bottom of any user-written



6 * S-Function so as to enable the static functions within it



7 * to be registered at startup.  Also, the macro S_FUNCTION_NAME



8 * should be declared within the S-Function to specify the



9 * entry point name--that is the name by which the S-Function



10 * is to be known to the outside world.



11 *



12 * Copyright 1994-2005 The MathWorks, Inc.



13 */

25  ssSetmdlInitializeConditionsLevel1(S, mdlInitializeConditions);



26  ssSetmdlOutputsLevel1(S, mdlOutputs); 



27  ssSetmdlUpdateLevel1(S, mdlUpdate); 



28  ssSetmdlDerivativesLevel1(S, mdlDerivatives);



29  ssSetmdlTerminate(S, mdlTerminate);



30#else



31  /* user level 2 */



32



33  ssSetmdlInitializeSizes(S, mdlInitializeSizes);



34  ssSetmdlInitializeSampleTimes(S, mdlInitializeSampleTimes); 



35

37    ssSetmdlInitializeConditions(S, mdlInitializeConditions);

88# endif

92# endif



93



94# if defined(MDL_CLEANUP_RUNTIME_RESOURCES)
