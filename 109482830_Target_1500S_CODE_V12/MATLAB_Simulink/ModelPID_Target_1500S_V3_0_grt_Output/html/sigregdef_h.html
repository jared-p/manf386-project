25#define SLREG_NO_FLAGS (0)

27#define SLREG_AVAIL           (1U << 0) 



28#define SLREG_REUSED          (1U << 1)



29#define SLREG_FCNCALL         (1U << 2)



30#define SLREG_MERGE           (1U << 3)



31#define SLREG_GROUND          (1U << 4) 



32#define SLREG_FRAME           (1U << 5)  /* opt exclude flag-see below          */



33#define SLREG_WIDE_FRAME      (1U << 6)  /* opt exclude flag-see below          */



34#define SLREG_MATRIX          (1U << 7)  /* opt exclude flag-see below          */



35#define SLREG_COMPLEX         (1U << 8)  /* opt exclude flag-see below          */



36#define SLREG_ALLOW_REUSED    (1U << 9)  /* opt un-exclude flag - see below     */



37#define SLREG_ACTION          (1U << 10) 



38#define SLREG_CONDEXEC        (1U << 11) /* signal is only conditional executed */



39#define SLREG_EXTMODE_UNAVAIL (1U << 12) /* signal is not available in extmode  */



40#define SLREG_ELIMINATED      (1U << 13) /* dead branch elimination             */



41#define SLREG_INCLUDE_GRND    (1U << 14) /* opt include ground regions          */



42#define SLREG_VIRTSIGCNV      (1U << 15) /* signal connected to signal type 



43                                          * conversion such as frame conversion



44                                          */



45#define SLREG_VARDIMS         (1U << 16) 



46#define SLREG_MESSAGES        (1U << 17)



47#define SLREG_DES_STAT        (1U << 18)



48#define SLREG_STRING          (1U << 19)

50#define IS_BIT_SET(uword32,bit) ((uword32 & bit) != 0)

52/* region node */



53typedef struct SL_SigRegion_tag {



54#if !defined(S_FUNCTION_NAME) && !defined(BUILDING_SL_LOGGING)



55    slPort *portObj;        /* output port associated with reg */



56#else



57    void *portObj;



58#endif



59    void         *data;     /* pointer to the data              */



60    int          startIdx;  /* starting element of the region (i.e., 'data'



61                             * points to the 'startIdx' element of the port)



62                             */



63    int          nEls;      /* number of elements in region     */



64    bool         entireSig; /* is this region the whole sig?    */



65    unsigned int status;    /* bit field: see SLREG_* status



66                             * vals above */



67



68    int index; /* The i'th region in the original map.  So, if a mux



69                * has three sigs coming in, a,b,c, a is index 0, b 1,



70                * and c 3.  If the regions were an array instead of



71                * a linked list this would be the array index. 



72                * BE CAREFUL: If regions are removed due to being



73                * duplicate or unavailable, the index field is not



74                * touched up, so the index field always remains what



75                * the original signal map for the composite signal



76                * represented.  I'm not sure if this is good behavior



77                * or not, but this is how it currently works:



78                * 1/16/04 HJT



79                */



80



81    int workInt;  /* A work integer.  Temp mem location for use when



82                   * shuffling regions.



83                   */



84



85    /*



86     * The following can be derived from the port, but since s-functions



87     * do not currently have direct access to the internal slPort *,



88     * we must put a copy of the required info directly into this struct.



89     */



90    int dType;



91    int dTypeSize;



92    int complexity;



93    int frameSize;



94



95    /*



96     *



97     */



98    bool discrete;   /*true if discrete in time or value (draw as stair)*/



99



100    /*



101     * Dimensions



102     *



103     * o Notes:



104     *    If not a frame (frameSize == 0)



105     *       dims[0...numDims], numDims = 1 for a scalar signal



106     *    If frame 



107     *       dims[0] = frameSize



108     *       dims[1] = number of columns



109     *    basically, the frame is treated as a matrix.



110     *



111    */



112    int numDims;



113    int *dims;



114



115    /* Cache var dims info in the sig region */



116    int     *currDims; 



117



118    struct SL_SigRegion_tag *next;



119    struct SL_SigRegion_tag *prev;



120



121    const char *name;



122



123    int unitId;



124    



125} SL_SigRegion;

128/*******************************************************************************



129 *                  SIGREGION: PUBLIC ACCESSORS                                *



130 ******************************************************************************/



131



132/*



133 * The number of elements in the region.



134 */



135#define gsr_nEls(reg) ((reg)->nEls)



136



137/*



138 * A pointer to the next region in the linked list. NULL if at the end of



139 * the list.



140 */



141#define gsr_NextReg(reg) ((reg)->next)



142



143/*



144 * A const char * pointer into the data.



145 */



146#define gsr_data(reg) ((char *)((reg)->data))



147



148/*



149 * Starting element of the region (i.e., 'data' points to the 'startIdx'



150 * element of the port)



151 */



152#define gsr_startIdx(reg) ((reg)->startIdx)



153



154/*



155 * Output port associated with region.



156 */



157#define gsr_portObj(reg) ((reg)->portObj)



158



159/* 



160 * The simulink data type identifier for the region.



161 */



162#define gsr_DataType(reg) ((reg)->dType)



163



164/*



165 * The number of bytes in one, non-complex data value.



166 */



167#define gsr_DataTypeSize(reg) ((reg)->dTypeSize)



168



169/*



170 * The simulink unit identifier for the region.



171 */



172#define gsr_Unit(reg) ((reg)->unitId)



173



174/*



175 * True if the data region contains complex elements.



176 */



177#define gsr_Complex(reg) ((reg)->complexity)



178



179/*



180 * The number of rows.  If not matrix, -1.



181 * NOTE: 



182 *  This access macro is deprecated and will only work correctly if 



183 *  gsr_NumDims <= 2 (i.e.: this macro does not support N-D signals N >2)



184 */



185#define gsr_M(reg) (((reg)->numDims <=1) ? -1 : (reg)->dims[0])



186



187/*



188 * The number of cols.  If not matrix, the sig width of the corresponding



189 * region. 



190 * NOTE: 



191 *  This access macro is deprecated and will only work correctly if 



192 *  gsr_NumDims <= 2 (i.e.: this macro does not support N-D signals N >2)



193 */



194#define gsr_N(reg) (((reg)->numDims <=1) ? (reg)->nEls : (reg)->dims[1])



195



196/*



197 * The number of dimensions. Can be greater than 2 for N-dimensional signals.



198 */



199#define gsr_NumDims(reg) ((reg)->numDims)



200



201/*



202 * An integer pointer to the dimensions vector. Use gsr_NumDims to determine



203 * the length of the dimensions array.



204 */



205#define gsr_Dims(reg) ((reg)->dims)



206



207/*



208 * The frameSize of the signal. Zero if not frame.



209 */



210#define gsr_FrameSize(reg) ((reg)->frameSize)



211



212/*



213 * The availability status of the region (SLREG_AVAIL if avail, else one



214 * of the status described above in the "Public Description" section.



215 */



216#define gsr_status(reg) ((reg)->status)



217



218#define gsr_index(reg) ((reg)->index)



219



220/*



221 * Var dims info - int pointer to current dims



222 */



223#define gsr_currDims(reg) ((reg)->currDims)



224



225



226#endif /* sigregdef_h */
