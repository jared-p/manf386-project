7// buffer passed into the Execute function. You can use the functions in

12// data types are converted to S7 data types in the write operation.  This 



13// class provides write access and inherits read access (from CODK_CpuReadData)

54void CODK_CpuReadWriteData::SetBuffer(long     nBytes, // Number of bytes in the data buffer



55                                      ODK_BYTE data[]) // Array pointer to the data buffer



56{



57      if (0 == data)



58      {



59            m_nBytes = 0;



60            m_LowIndex = 0;



61            m_HighIndex = 0;

63      else

67            m_HighIndex = -1;

72//



73// WriteUINT8 : Writes a generic 8-bit value to the data area



74//



75bool CODK_CpuReadWriteData::WriteUINT8(const long      byteOffset, // Index into the data buffer



76                                       const ODK_UINT8 value)      // Value to write into the buffer



77{



78    // Verify offset is within data area



79    if ((byteOffset < 0) || ((byteOffset + ODK_SIZEBIT8) > m_nBytes))



80    {



81        return false;



82    }



83



84    // Update highest index accessed



85    if ((byteOffset + ODK_SIZEBIT8 - 1) > m_HighIndex)



86    {



87        m_HighIndex = byteOffset + ODK_SIZEBIT8 - 1;



88    }



89



90    // Update lowest index accessed



91    if (byteOffset < m_LowIndex)



92    {



93        m_LowIndex = byteOffset;



94    }



95



96    // Convert and write the data



97    m_data[byteOffset] = (ODK_UINT8)value; 



98



99    return true;



100}

209//



210// WriteS7BOOL : Writes a boolean value to the data area



211//



212bool CODK_CpuReadWriteData::WriteS7BOOL(const long         byteOffset, // Index into the data buffer



213                                        const unsigned int bitNo,      // Number of the bit to write (index from right to left)



214                                        const bool         value)      // Value of the bit to write



215{



216    // verify correct bit range



217    if (bitNo > 7)



218    {



219        return false;



220    }



221



222    // Bit indexing: least significant bit is 0 (i.e. go right to left)



223



224    ODK_UINT8 temp;  // variable for the byte's current value



225



226    // Get the current byte so the other bit values can be maintained



227    if (ReadUINT8(byteOffset, temp) == false)



228    {



229        return false;



230    }



231



232    // Set the bit in the byte to be written



233    if (true == value)



234    {



235        temp = static_cast<ODK_UINT8>(temp | (1U << bitNo));



236    }



237    else



238    {



239        temp = static_cast<ODK_UINT8>(temp & (~(1U << bitNo)));



240    }



241    // Write the byte



242    return WriteUINT8(byteOffset, temp);



243}

301//



302// WriteS7DTL : Writes a Date and Time as DTL to the data area



303//



304bool CODK_CpuReadWriteData::WriteS7DTL(long          byteOffset, // Offset to begin reading



305                                       const ODK_DTL &dtl)       // DTL structure to write



306{



307	// Verify offset is within data area



308    if ((byteOffset < 0) || ((byteOffset + static_cast<long>(sizeof(dtl))) > m_nBytes))



309    {



310        return false;



311    }



312



313	// Check content of DTL



314	if (!CheckS7DTL(dtl))



315	{



316        return false;



317	}



318



319    // year



320    WriteUINT16(byteOffset, dtl.Year);



321    byteOffset += ODK_SIZEBIT16;



322    // month



323    WriteUINT8(byteOffset, dtl.Month);



324    byteOffset += ODK_SIZEBIT8;



325    // day



326    WriteUINT8(byteOffset, dtl.Day);



327    byteOffset += ODK_SIZEBIT8;



328    // weekday



329    WriteUINT8(byteOffset, dtl.Weekday);



330    byteOffset += ODK_SIZEBIT8;



331    // hour



332    WriteUINT8(byteOffset, dtl.Hour);



333    byteOffset += ODK_SIZEBIT8;



334    // minute



335    WriteUINT8(byteOffset, dtl.Minute);



336    byteOffset += ODK_SIZEBIT8;



337    // second



338    WriteUINT8(byteOffset, dtl.Second);



339    byteOffset += ODK_SIZEBIT8;



340    // nanosecond



341    WriteUINT32(byteOffset, dtl.Nanosecond);



342



343	return true;



344}

346//



347// WriteS7STRING : Writes a string to the data area



348//                 Truncate destination string, when source string is too long.



349//



350bool CODK_CpuReadWriteData::WriteS7STRING(const long      byteOffset, // Index into the data buffer



351                                          const ODK_CHAR* string)     // Pointer to a string buffer containing the string to write



352{



353    ODK_UINT8 maxLen = 0;



354    ODK_UINT8 curLen = 0;



355



356    // check parameters and read maxLen



357    if (0 == string



358     || ! ReadUINT8(byteOffset, maxLen))



359    {



360        return false;



361    }



362



363    // get the current size of the string



364    curLen = static_cast<ODK_UINT8>(strlen(string));



365



366	// get the value to write to



367    if (curLen > maxLen)



368    {



369        curLen = maxLen;



370    }



371



372    // verify offset not out of bounds



373    if (static_cast<long>(byteOffset + curLen + (2 * sizeof(ODK_CHAR))) > m_nBytes)



374    {



375        return false;



376    }



377



378    // set the current length



379    WriteUINT8(byteOffset + 1, curLen);



380



381	// Copy the string



382	memcpy(&(m_data[byteOffset + 2]), string, curLen);



383



384    return true;



385}
