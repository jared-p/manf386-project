153/* DataTypeMap structure */



154/* Members provide data type information of a signal or parameter */ 



155typedef struct rtwCAPI_DataTypeMap_tag {



156  const char_T *cDataName;   /* C language data type name                    */



157  const char_T *mwDataName;  /* MathWorks data type, typedef in rtwtypes.h   */



158  uint16_T      numElements; /* number of elements, 0 for non-structure data */



159  uint16_T      elemMapIndex;/* index into the ElementMap, gives Bus Info    */



160  uint16_T      dataSize;    /* data size in Bytes                           */



161  uint8_T       slDataId;    /* enumerated data type from simstruc_types.h   */



162  unsigned int  isComplex:1; /* is the data type complex (1=Complex, 0=Real) */



163  unsigned int  isPointer:1; /* is data accessed Via Pointer (1=yes, 0= no)  */



164} rtwCAPI_DataTypeMap;



165



166/* Notes on rtwCAPI_DataTypeMap:



167 * cDataName  - The ANSI C equivalent data type. 



168 *              For fixed point data



169 *                  cDataName = the native integer equivalent



170 *              For Complex data or non-virtual structures, 



171 *                  cDataName = "struct" 



172 * mwDataName - RTW defined data type. Typedef can be found in rtwtypes.h



173 *              For fixed point data, 



174 *                  mwDataName = the integer type in which the data is stored.



175 *                  Use rtwCAPI_FixPtMap to get slope & bias of the data.



176 * numElements - Number of elements/members in the data type. If the data type 



177 *              represents a non-virtual structure (for e.g, bus structure),



178 *              this field gives the number of elements in the structure. 



179 *              For non-structure data type, the default value is 0



180 *              Complex data types are treated as non-structures



181 * elemMapIndex - Index into the rtwCAPI_ElementMap which maps each of the



182 *              element/member of a structure data type.



183 * slDataId   - Enumerated Simulink Data type defined in simstruc_types.h.



184 *              The enumerations are



185 *                 o SS_DOUBLE



186 *                 o SS_SINGLE



187 *                 o SS_INT8



188 *                 o SS_UINT8



189 *                 o SS_INT16



190 *                 o SS_UINT16



191 *                 o SS_INT32



192 *                 o SS_UINT32



193 *                 o SS_BOOLEAN



194 *                 o SS_ENUM_TYPE  (for enumerated data types)



195 *                 o SS_STRUCT     (for non-virtual structures)



196 *              For fixed point data: 



197 *                  slDataId = enumeration corresponding to the integer type 



198 *                             in which the data is stored



199 *              For Complex data: 



200 *                  slDataId = enumeration corresponding to the data type of 



201 *                             the real part.



202 * isComplex  - is the data type complex (1=Complex, 0=Real) 



203 * isPointer  - is data accessed Via Pointer (1=yes, 0= no)



204 */



205
