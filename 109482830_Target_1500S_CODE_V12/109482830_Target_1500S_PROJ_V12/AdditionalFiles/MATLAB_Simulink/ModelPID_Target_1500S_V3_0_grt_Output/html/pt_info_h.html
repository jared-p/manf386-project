11#ifndef __PT_INFO__



12#define __PT_INFO__



13



14#include "rtwtypes.h"



15



16typedef enum {



17    rt_SCALAR,



18    rt_VECTOR,



19    rt_MATRIX_ROW_MAJOR,



20    rt_MATRIX_COL_MAJOR,



21    rt_MATRIX_COL_MAJOR_ND



22} ParamClass;



23



24typedef enum {



25    rt_SL_PARAM,



26    rt_SF_PARAM,



27    rt_SHARED_PARAM



28} ParamSource;



29



30typedef struct ParameterTuning_tag {



31    ParamClass  paramClass;   /* Class of parameter               */



32    int_T       nRows;        /* Number of rows                   */



33    int_T       nCols;        /* Number of columns                */



34    int_T       nDims;        /* Number of dimensions             */



35    int_T       dimsOffset;   /* Offset into dimensions vector    */



36    ParamSource source;       /* Source of parameter              */



37    uint_T      dataType;     /* data type enumeration            */



38    uint_T      numInstances; /* Number of parameter instances    */



39    int_T       mapOffset;    /* Offset into map vector           */



40} ParameterTuning;



41



42typedef struct BlockTuning_tag {



43    const char_T      *blockName;   /* Block name                       */



44    const char_T      *paramName;   /* Parameter name                   */



45    ParameterTuning    ptRec;       /* Parameter tuning record          */



46} BlockTuning;



47



48typedef struct VariableTuning_tag {



49    const char_T *varName;    /* Variable name                    */



50    ParameterTuning ptRec;    /* Parameter tuning record          */



51} VariableTuning;



52



53#define ptinfoGetClass(pt)            (pt)->paramClass



54#define ptinfoGetNumRows(pt)          (pt)->nRows



55#define ptinfoGetNumCols(pt)          (pt)->nCols



56#define ptinfoGetNumDimensions(pt)    (pt)->nDims



57#define ptinfoGetSource(pt)           (pt)->source



58#define ptinfoGetDataTypeEnum(pt)     (pt)->dataType



59#define ptinfoGetNumInstances(pt)     (pt)->numInstances



60#define ptinfoGetParametersOffset(pt) (pt)->mapOffset



61#define ptinfoGetDimensionsOffset(pt) (pt)->dimsOffset



62



63#endif  /* __PT_INFO__ */
