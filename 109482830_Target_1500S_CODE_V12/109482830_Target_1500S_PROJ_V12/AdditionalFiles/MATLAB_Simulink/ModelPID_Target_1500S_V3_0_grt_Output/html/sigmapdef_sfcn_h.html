47/*



48 * A high-level overview of the signal list is provided here.  For more details,



49 * feel free to examine the data structures defined above.  More information



50 * is also provided for the individual access macros defined below.



51 *



52 * WARNING:



53 *  DO NOT DIRECTLY ACCESS ANY OF THESE DATA STRUCTURES.  THEY MAY CHANGE



54 *  IN FUTURE RELEASES.  USE THE ACCESS MACROS PROVIDED BELOW.  SEE



55 *  <matlabroot>/simulink/src/barplot.c FOR A WORKING EXAMPLE OF DATA ACCESS



56 *  VIA SIGNAL MAPS.



57 *



58 *



59 * A signal list is an array of sigInfo structures:



60 *  -----------------------------------



61 *  | graphical port | graphical port |



62 *  | signal map     | signal map     |



63 *  -----------------------------------



64 *



65 *      graphical port:



66 *          A pointer to the output port associated with a signal.  It is



67 *          graphical in the sense that it may or may not map back to the



68 *          actual outport from which the data originates.   Consider:



69 *



70 *          -----        --------



71 *          | A |>------>|      |



72 *          -----        |      |



73 *          -----        |  C   |>---------->



74 *          | B |>------>|      |



75 *          -----        |------|



76 *                         mux



77 *



78 *          For the mux block, the 'graphical' port is C.  The two 'actual'



79 *          ports are A and B.



80 *



81 *      signal map:



82 *          A linked list of contiguous regions of memory corresponding to



83 *          the graphical signal.  Assuming that A and B above are 



84 *          non-virtual blocks (e.g., gains, constants, etc, but not mux



85 *          demux, etc), then the signal C has two contiguous regions.



86 *



87 *          The signal map looks like:



88 *              



89 *              ----------------   --> ----------------   --> NULL



90 *              | region info A|   |   | region info B|   |



91 *              | next region--|----   | next region--|----



92 *              ----------------       ---------------- 



93 *



94 *          Note that if we create a signal list for A, the graphical port



95 *          and the actual port are an identity mapping.  The resulting



96 *          signal map consists would consist of only 1 region (assuming that



97 *          the output of A is contiguous).



98 *              



99 *          The regions are essentially pointers into the Simulink memory areas



100 *          that are used for block input and output.  Here is some of the info 



101 *          contained in the regions.  Other info is also available.  See macro



102 *          definitions at the bottom of this file.



103 *



104 *              data: a void pointer to the data in this region



105 *              nEls: number of elements in region



106 *              



107 *              status: a flag that indicates whether or not the region is



108 *                      available for access.  When regions are not available,



109 *                      the 'data' field is NULL.  Status can take on the



110 *                      following values:



111 *                  



112 *                          SLREG_AVAIL:



113 *                              Data is available for access.



114 *



115 *                          ====================================================



116 *                          The following types of regions are excluded by



117 *                          default from the signal list.



118 *                          ====================================================



119 *



120 *                          SLREG_REUSED:



121 *                              The output memory area for this signal is being



122 *                              shared with the output from another port.  This



123 *                              signal can not be accessed (click 'Disable



124 *                              optimized block I/O storage' on the Diagnostics



125 *                              page of the Simulation Parameter Dialog box



126 *                              to disable the sharing of block I/O memory).



127 *



128 *                          SLREG_FCNCALL:



129 *                              The data corresponds to a fcn-call connection.



130 *                              There is no data to view.



131 *



132 *                          SLREG_MERGE:



133 *                              This signal is being merged.  This is a special



134 *                               case of a buffer being re-used.  The



135 *                               signal cannot be accessed.



136 *



137 *                          SLREG_ACTION:



138 *                              The data corresponds to an action signal, this



139 *                              data can not be viewed. 



140 *                             



141 *                          ====================================================



142 *                          The following types of regions are optionally



143 *                          excluded from the signal list.  See the excludeFlags



144 *                          arg of ssCallSigListCreateFcn().



145 *



146 *                          If you are not excluding a given region type, the 



147 *                          region status will be SLREG_AVAIL.  Otherwise, if



148 *                          the excluded region type, SLREG_xxx, is encountered,



149 *                          the region status will be SLREG_xxx.



150 *                          ====================================================



151 *



152 *                          SLREG_FRAME:



153 *                              The data region corresponds to a frame (a time



154 *                              series of data points).  If your application



155 *                              does not support frames, include SLREG_FRAME



156 *                              as part of the excludeFlags to ensure that



157 *                              all frame data is excluded from the sigList.



158 *



159 *                          SLREG_WIDE_FRAME:



160 *                              The data region corresponds to a wide frame, 



161 *                              size greated than 1. If your application



162 *                              does not support wide frames, include 



163 *                              SLREG_WIDE_FRAME as part of the excludeFlags to 



164 *                              ensure that all wide frame data is excluded from the 



165 *                              sigList.



166 *



167 *                          SLREG_MATRIX:



168 *                              The data region corresponds to a matrix.  If



169 *                              your application does not support matrices,



170 *                              include SLREG_MATRIX as part of the 



171 *                              excludeFlags to ensure that all matrices are



172 *                              excluded from the sigList.



173 *



174 *                          SLREG_COMPLEX:



175 *                              The data region corresponds to complex data.  If



176 *                              your application does not support complex data,



177 *                              include SLREG_COMPLEX as part of the 



178 *                              excludeFlags to ensure that all complex data is



179 *                              excluded from the sigList.



180 *



181 *                          ====================================================



182 *                          The following overrides the default exclusions



183 *                          of the specified type of regions.



184 *                          ====================================================



185 *



186 *                          SLREG_ALLOW_REUSED



187 *                              "un-exludes" reused regions (which are excluded



188 *                              by default.  As described above in the



189 *                              SLREG_REUSED section, this can lead to



190 *                              unexpected results, unless the block that is



191 *                              requesting the signal access is hardwired into



192 *                              the diagram.



193 *



194 *                              In general, if a block has input ports, this



195 *                              flag should be specified (assuming that the 



196 *                              sigmaps being created correspond to its input



197 *                              ports).



198 *



199 *              dType:      The Simulink data type id for the data.



200 *              dTypeSize:  Number of bytes in 1 non-complex element.



201 *              complexity: true if the data values are complex



202 *              m:          if a matrix, the number of rows (-1 if not matrix)



203 *              n:          if a matrix, the number of cols (-1 if not matrix)



204 *



205 *



206 *  In general, the following snippet of code can be used to walk the elements



207 *  of a signal:



208 *      SL_SigRegion *sigReg = gsl_FirstReg(sigList,i);



209 *       



210 *      do {



211 *         int nEls = gsr_nEls(sigReg);



212 *



213 *          if (nEls > 0) {



214 *              int        el;



215 *              int        dType   = gsr_DataType(sigReg);



216 *              int        elSize  = gsr_DataTypeSize(sigReg);



217 *              const char *data   = gsr_data(sigReg);



218 *



219 *              for (el=0; el<nEls; el++) {



220 *                  data += elSize;



221 *              }



222 *          }



223 *      } while((sigReg = gsr_NextReg(sigReg)) != NULL);



224 */

254/*



255 * Returns true if the signal is a tie wrap.  This means that the signal map



256 * consists of either:



257 *  o one entire signal (i.e., the output of 1 non-virtual block)



258 *  o a simple bundling together of multiple entire signals



259 *



260 *  Generally the only way to get a non-tie-wrapped signal is to mix the



261 *  elements of existing signals or re-shuffle them via the use of the selector,



262 *  mux and demux blocks.



263 *



264 *



265 *      -----      -----                               --------



266 *      | 1 | ---->|   |           -----               |      |



267 *      -----      |   |           |   |        ------>|      |



268 *     constant    |   | tie wrap  |   |------  |      |      |  not a tie wrap  



269 *      -----      |   |---------->|   |     |  |      |      |--------------->



270 *      | 1 | ---->|   |           |   |  ---|--|      |      | elements got 



271 *      -----      |   |           |   |--|  |-------->|      |   shuffled



272 *     constant    -----           |   |               |      |



273 *                   mux           -----               --------



274 *                                 demux                 mux



275 *



276 *



277 */



278#define gsl_TieWrap(S, sigList, lstIdx) \



279           ssCallSigListGetIfTieWrapFcn(S, sigList, lstIdx)
