20typedef struct ModelMappingInfo_tag {

22  struct {



23    BlockIOSignals const *blockIOSignals;    /* Block signals map             */



24    uint_T               numBlockIOSignals;  /* Num signals in map            */



25  } Signals;

27  /* parameter tuning */



28  struct {



29    BlockTuning const    *blockTuning;       /* Block parameters map          */



30    VariableTuning const *variableTuning;    /* Variable parameters map       */



31    void * const         *parametersMap;     /* Parameter index map           */



32    uint_T const         *dimensionsMap;     /* Dimensions index map          */



33    uint_T                numBlockTuning;    /* Num block parameters in map   */



34    uint_T                numVariableTuning; /* Num variable parameter in map */



35  } Parameters;



36} ModelMappingInfo;
