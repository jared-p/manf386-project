45typedef struct rtwCAPI_ModelMappingInfo_tag rtwCAPI_ModelMappingInfo;



46typedef struct rtwCAPI_ModelMappingStaticInfo_tag rtwCAPI_ModelMappingStaticInfo;

48/* ModelMappingStaticInfo */



49struct rtwCAPI_ModelMappingStaticInfo_tag {



50    /* signals */



51    struct {



52        rtwCAPI_Signals const *signals;     /* Signals Array */



53        uint_T                numSignals;   /* Num Signals   */



54        rtwCAPI_Signals const *rootInputs;  /* Root Inputs array */



55        uint_T               numRootInputs; /* Num Root Inputs  */



56        rtwCAPI_Signals const *rootOutputs; /* Root Outputs array */



57        uint_T               numRootOutputs;/* Num Root Outputs  */



58    } Signals;



59



60    /* parameters */



61    struct {



62        rtwCAPI_BlockParameters const *blockParameters;   /* Block parameters Array   */



63        uint_T                       numBlockParameters; /* Num block parameters     */



64        rtwCAPI_ModelParameters const *modelParameters;   /* Model parameters Array*/



65        uint_T                       numModelParameters; /* Num Model parameters  */



66    } Params;



67



68    /* states */



69    struct {



70        rtwCAPI_States const *states;   /* States array */



71        uint_T               numStates; /* Num States   */



72    } States;



73



74    /* Static maps */



75    /* datatypes, dimensions, fixed point, structure elements, sample times  */



76    struct {



77        rtwCAPI_DataTypeMap   const   *dataTypeMap;       /* Data Type Map          */



78        rtwCAPI_DimensionMap  const   *dimensionMap;      /* Data Dimension Map     */



79        rtwCAPI_FixPtMap      const   *fixPtMap;          /* Fixed Point Map        */



80        rtwCAPI_ElementMap    const   *elementMap;        /* Structure Element map  */



81        rtwCAPI_SampleTimeMap const   *sampleTimeMap;     /* Sample Times Map       */



82        uint_T        const   *dimensionArray;    /* Dimension Array        */



83    } Maps;



84



85    /* TargetType - string specifying the intended target of the generated  *



86     * C-API.                                                               *



87     * targetType = "float"   - target supports floats and integer code     *



88     *            = "integer" - target supports integer only code           */



89    char_T const     *targetType;



90



91    /* for internal use */



92    uint32_T modelChecksum[4];



93    rtwCAPI_ModelMapLoggingStaticInfo const *staticLogInfo;



94    size_t	 rtpSize;



95    /* If this instance in a protected model*/



96    boolean_T isProtectedModel;



97};



98



99/* ModelMappingInfo */



100struct rtwCAPI_ModelMappingInfo_tag {



101    /* ModelMappingInfo version */



102    uint8_T                 versionNum;



103



104    /* Reference to static model data, all model instances share this map   */



105    rtwCAPI_ModelMappingStaticInfo *staticMap;



106



107    /* Instance specific Maps, each model instance has a unique InstanceMap */



108    struct {



109        const char*                path;             /* Path to this instance */



110        char*                      fullPath;



111        void**                     dataAddrMap;      /* Data Address map      */



112        rtwCAPI_ModelMappingInfo** childMMIArray;    /* array of child MMI    */



113        uint_T                     childMMIArrayLen; /* Number of child MMIs  */



114        int_T                      contStateStartIndex; 



115



116        /* for internal use */



117        rtwCAPI_ModelMapLoggingInstanceInfo *instanceLogInfo;



118        int32_T**                      vardimsAddrMap;   /* Vardims Address map   */



119        void*			       rtpAddress;



120        RTWLoggingFcnPtr* RTWLoggingPtrs;   /* MatFile logging information */



121    } InstanceMap;



122};
